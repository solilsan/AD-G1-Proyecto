create TABLE CLIENTES(
  DNI VARCHAR(50) PRIMARY KEY,
  NOMBRE VARCHAR(50) NOT NULL,
  APELLIDO VARCHAR(50) NOT NULL,
  FECHA_NAC DATE NOT NULL,
  PROFESION  VARCHAR(50) NOT NULL,
  ESTADO  VARCHAR(50) NOT NULL
);

create TABLE EMPLEADOS(
  DNI VARCHAR(50) PRIMARY KEY,
  NOMBRE VARCHAR(50) NOT NULL,
  APELLIDO VARCHAR(50) NOT NULL,
  FECHA_NAC DATE NOT NULL,
  F_CONTRATACION DATE NOT NULL,
  NACIONALIDAD VARCHAR(50) NOT NULL,
  CARGO  VARCHAR(50) NOT NULL,
  PASSWORD VARCHAR(50) NOT NULL,
  ESTADO  VARCHAR(50) NOT NULL
);

create TABLE VISITAS(
  ID INTEGER PRIMARY KEY AUTOINCREMENT,
  NOMBRE VARCHAR(50) NOT NULL,
  N_MAX_CLI INTEGER(10) NOT NULL,
  PUNTO_PARTIDA VARCHAR(50) NOT NULL,
  CURSO_ACADEMICO VARCHAR(50) NOT NULL,
  TEMATICA VARCHAR(50) NOT NULL,
  COSTE float(10) NOT NULL,
  ESTADO VARCHAR(50) NOT NULL,
  DNI_EMPLEADO VARCHAR(50) NOT NULL,
  FOREIGN KEY(DNI_EMPLEADO) REFERENCES EMPLEADOS(dni)
);

create TABLE V_GUIADA(
  ID_VISITA INTEGER NOT NULL,
  DNI_CLI VARCHAR NOT NULL,
  FOREIGN KEY(ID_VISITA) REFERENCES VISITAS(ID),
  FOREIGN KEY(DNI_CLI) REFERENCES CLIENTES(DNI)
);
